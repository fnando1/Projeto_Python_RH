-- ============================================================ --  SISTEMA DE RH — Script DDL MySQL --  Gerado em: 2025 -- ============================================================

CREATE DATABASE IF NOT EXISTS sistema_rh DEFAULT CHARACTER SET utf8mb4 DEFAULT COLLATE utf8mb4_unicode_ci;

USE sistema_rh;

-- ============================================================ --  1. PERFIL_ACESSO  (os 4 tipos de login) -- ============================================================
CREATE TABLE perfil_acesso (
id_perfil       INT          NOT NULL AUTO_INCREMENT,
nome            ENUM(‘DIRETOR’,‘GERENTE_RH’,‘AUXILIAR_RH’,‘CANDIDATO’) NOT NULL UNIQUE,
descricao       TEXT,
nivel_acesso    TINYINT      NOT NULL  -- 1=mais alto,
PRIMARY KEY (id_perfil)
) ENGINE=InnoDB;

-- Dados iniciais dos perfis
INSERT INTO perfil_acesso (nome, descricao, nivel_acesso) VALUES
(‘DIRETOR’,      ‘Acesso total ao sistema’,           1),
(‘GERENTE_RH’,   ‘Gerencia processos de RH’,           2),
(‘AUXILIAR_RH’,  ‘Operações básicas de RH’,            3),
(‘CANDIDATO’,    ‘Acesso ao formulário e candidatura’, 4);

-- ============================================================ --  2. PESSOA  (formulário de cadastro público) -- ============================================================
CREATE TABLE pessoa (
id_pessoa       INT              NOT NULL AUTO_INCREMENT,
nome_completo   VARCHAR(150)     NOT NULL,
cpf             VARCHAR(14)      NOT NULL UNIQUE,
email           VARCHAR(120)     NOT NULL UNIQUE,
telefone        VARCHAR(20),
data_nascimento DATE             NOT NULL,
genero          ENUM(‘M’,‘F’,‘NB’,‘NAO_INFORMADO’) DEFAULT ‘NAO_INFORMADO’,
endereco        TEXT,
foto_perfil     VARCHAR(255),
criado_em       TIMESTAMP        DEFAULT CURRENT_TIMESTAMP,
PRIMARY KEY (id_pessoa)
) ENGINE=InnoDB;

-- ============================================================ --  3. USUARIO  (controle de login) -- ============================================================
CREATE TABLE usuario (
id_usuario      INT          NOT NULL AUTO_INCREMENT,
id_pessoa       INT          NOT NULL UNIQUE,
login           VARCHAR(80)  NOT NULL UNIQUE,
senha_hash      VARCHAR(255) NOT NULL,
id_perfil       INT          NOT NULL,
ativo           TINYINT(1)   NOT NULL DEFAULT 1,
ultimo_acesso   TIMESTAMP    NULL,
token_reset     VARCHAR(255),
criado_em       TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
PRIMARY KEY (id_usuario),
CONSTRAINT fk_usuario_pessoa  FOREIGN KEY (id_pessoa)  REFERENCES pessoa(id_pessoa),
CONSTRAINT fk_usuario_perfil  FOREIGN KEY (id_perfil)  REFERENCES perfil_acesso(id_perfil)
) ENGINE=InnoDB;

-- ============================================================ --  4. DEPARTAMENTO -- ============================================================
CREATE TABLE departamento (
id_departamento INT         NOT NULL AUTO_INCREMENT,
nome            VARCHAR(100) NOT NULL,
sigla           VARCHAR(10),
id_gerente      INT         NULL,   -- FK para funcionario (definida após)
id_dept_pai     INT         NULL,   -- hierarquia de departamentos
ativo           TINYINT(1) NOT NULL DEFAULT 1,
PRIMARY KEY (id_departamento),
CONSTRAINT fk_dept_pai  FOREIGN KEY (id_dept_pai) REFERENCES departamento(id_departamento)
) ENGINE=InnoDB;

-- ============================================================ --  5. CARGO -- ============================================================
CREATE TABLE cargo (
id_cargo         INT            NOT NULL AUTO_INCREMENT,
titulo           VARCHAR(100)   NOT NULL,
id_departamento  INT            NOT NULL,
descricao        TEXT,
salario_base     DECIMAL(10,2),
nivel_hierarquico VARCHAR(30),
cbo              VARCHAR(10),
PRIMARY KEY (id_cargo),
CONSTRAINT fk_cargo_depto FOREIGN KEY (id_departamento) REFERENCES departamento(id_departamento)
) ENGINE=InnoDB;

-- ============================================================ --  6. FUNCIONARIO -- ============================================================
CREATE TABLE funcionario (
id_funcionario  INT            NOT NULL AUTO_INCREMENT,
id_pessoa       INT            NOT NULL UNIQUE,
matricula       VARCHAR(20)    NOT NULL UNIQUE,
id_cargo        INT            NOT NULL,
id_departamento INT            NOT NULL,
data_admissao   DATE           NOT NULL,
data_demissao   DATE           NULL,
salario_atual   DECIMAL(10,2)  NOT NULL,
tipo_contrato   ENUM(‘CLT’,‘PJ’,‘ESTAGIO’,‘TEMPORARIO’) NOT NULL DEFAULT ‘CLT’,
id_superior     INT            NULL,
status          ENUM(‘ATIVO’,‘INATIVO’,‘AFASTADO’,‘FERIAS’) NOT NULL DEFAULT ‘ATIVO’,
PRIMARY KEY (id_funcionario),
CONSTRAINT fk_func_pessoa   FOREIGN KEY (id_pessoa)       REFERENCES pessoa(id_pessoa),
CONSTRAINT fk_func_cargo    FOREIGN KEY (id_cargo)        REFERENCES cargo(id_cargo),
CONSTRAINT fk_func_depto    FOREIGN KEY (id_departamento) REFERENCES departamento(id_departamento),
CONSTRAINT fk_func_superior FOREIGN KEY (id_superior)     REFERENCES funcionario(id_funcionario)
) ENGINE=InnoDB;

-- Agora adiciona FK do gerente no departamento
ALTER TABLE departamento ADD CONSTRAINT fk_depto_gerente FOREIGN KEY (id_gerente) REFERENCES funcionario(id_funcionario);

-- ============================================================ --  7. VAGA -- ============================================================
CREATE TABLE vaga (
id_vaga          INT            NOT NULL AUTO_INCREMENT,
id_cargo         INT            NOT NULL,
id_departamento  INT            NOT NULL,
id_criador       INT            NOT NULL,  -- FK para usuario
titulo           VARCHAR(150)   NOT NULL,
descricao        TEXT,
requisitos       TEXT,
qtd_vagas        INT            NOT NULL DEFAULT 1,
tipo             ENUM(‘CLT’,‘PJ’,‘ESTAGIO’,‘TEMPORARIO’),
salario_ofertado DECIMAL(10,2),
data_abertura    DATE           NOT NULL,
data_encerramento DATE,
status           ENUM(‘ABERTA’,‘ENCERRADA’,‘SUSPENSA’,‘PREENCHIDA’) NOT NULL DEFAULT ‘ABERTA’,
PRIMARY KEY (id_vaga),
CONSTRAINT fk_vaga_cargo  FOREIGN KEY (id_cargo)        REFERENCES cargo(id_cargo),
CONSTRAINT fk_vaga_depto  FOREIGN KEY (id_departamento) REFERENCES departamento(id_departamento),
CONSTRAINT fk_vaga_criador FOREIGN KEY (id_criador)     REFERENCES usuario(id_usuario)
) ENGINE=InnoDB;

-- ============================================================ --  8. CANDIDATURA -- ============================================================
CREATE TABLE candidatura (
id_candidatura  INT            NOT NULL AUTO_INCREMENT,
id_usuario      INT            NOT NULL,
id_vaga         INT            NOT NULL,
data_candidatura TIMESTAMP     DEFAULT CURRENT_TIMESTAMP,
status          ENUM(‘INSCRITO’,‘EM_TRIAGEM’,‘ENTREVISTA’,‘APROVADO’,‘REPROVADO’,‘DESISTIU’) NOT NULL DEFAULT ‘INSCRITO’,
curriculo_url   VARCHAR(255),
carta_motivacao TEXT,
nota_triagem    DECIMAL(4,2),
id_avaliador    INT            NULL,
obs_avaliador   TEXT,
PRIMARY KEY (id_candidatura),
UNIQUE KEY uq_candidatura (id_usuario, id_vaga),
CONSTRAINT fk_cand_usuario    FOREIGN KEY (id_usuario)   REFERENCES usuario(id_usuario),
CONSTRAINT fk_cand_vaga       FOREIGN KEY (id_vaga)      REFERENCES vaga(id_vaga),
CONSTRAINT fk_cand_avaliador  FOREIGN KEY (id_avaliador) REFERENCES usuario(ENGINE=InnoDB;

-- ============================================================ --  9. HISTORICO_SALARIAL -- ============================================================
CREATE TABLE historico_salarial (
id_historico     INT            NOT NULL AUTO_INCREMENT,
id_funcionario   INT            NOT NULL,
salario_anterior DECIMAL(10,2),
salario_novo     DECIMAL(10,2)  NOT NULL,
motivo           VARCHAR(150),
id_aprovador     INT            NULL,
data_alteracao   DATE           NOT NULL,
PRIMARY KEY (id_historico),
CONSTRAINT fk_hist_func      FOREIGN KEY (id_funcionario) REFERENCES funcionario(id_funcionario),
CONSTRAINT fk_hist_aprovador FOREIGN KEY (id_aprovador)   REFERENCES usuario(id_usuario)
) ENGINE=InnoDB;

-- ============================================================ --  10. AVALIACAO_DESEMPENHO -- ============================================================
CREATE TABLE avaliacao_desempenho (
id_avaliacao         INT           NOT NULL AUTO_INCREMENT,
id_funcionario       INT           NOT NULL,
id_avaliador         INT           NOT NULL,
periodo_inicio       DATE          NOT NULL,
periodo_fim          DATE          NOT NULL,
nota_tecnica         DECIMAL(4,2),
nota_comportamental  DECIMAL(4,2),
nota_geral           DECIMAL(4,2),
comentarios          TEXT,
criado_em            TIMESTAMP     DEFAULT CURRENT_TIMESTAMP,
PRIMARY KEY (id_avaliacao),
CONSTRAINT fk_aval_func      FOREIGN KEY (id_funcionario) REFERENCES funcionario(id_funcionario),
CONSTRAINT fk_aval_avaliador FOREIGN KEY (id_avaliador)   REFERENCES funcionario(id_funcionario)
) ENGINE=InnoDB;

-- ============================================================ --  11. TREINAMENTO -- ============================================================
CREATE TABLE treinamento (
id_treinamento  INT           NOT NULL AUTO_INCREMENT,
nome            VARCHAR(150)  NOT NULL,
descricao       TEXT,
modalidade      ENUM(‘PRESENCIAL’,‘EAD’,‘HIBRIDO’),
carga_horaria   INT,
data_inicio     DATE,
data_fim        DATE,
custo           DECIMAL(10,2),
PRIMARY KEY (id_treinamento)
) ENGINE=InnoDB;

-- ============================================================ --  12. FUNCIONARIO_TREINAMENTO  (tabela de junção N:M) -- ============================================================
CREATE TABLE funcionario_treinamento (
id              INT           NOT NULL AUTO_INCREMENT,
id_funcionario  INT           NOT NULL,
id_treinamento  INT           NOT NULL,
status          ENUM(‘INSCRITO’,‘EM_ANDAMENTO’,‘CONCLUIDO’,‘CANCELADO’) DEFAULT ‘INSCRITO’,
nota_conclusao  DECIMAL(4,2),
data_conclusao  DATE,
certificado_url VARCHAR(255),
PRIMARY KEY (id),
UNIQUE KEY uq_func_trein (id_funcionario, id_treinamento),
CONSTRAINT fk_ft_func  FOREIGN KEY (id_funcionario) REFERENCES funcionario(id_funcionario),
CONSTRAINT fk_ft_trein FOREIGN KEY (id_treinamento) REFERENCES treinamento(id_treinamento)
) ENGINE=InnoDB;
